import java.util.Scanner; public class Main { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); String[] names = new String[10]; int[] scores = new int[10]; int totalScore = 0; int countAbove17 = 0; int countBelow15 = 0; // Get names and scores for (int i = 0; i < 10; i++) { System.out.print("Enter name " + (i + 1) + ": "); names[i] = scanner.nextLine(); System.out.print("Enter score for " + names[i] + ": "); scores[i] = scanner.nextInt(); scanner.nextLine(); // To consume the newline character after entering a number // Display the length of the name System.out.println("Length of " + names[i] + "'s name: " + names[i].length()); // Check the scores if (scores[i] > 17) { countAbove17++; System.out.println(names[i] + " has a score above 17: " + scores[i]); } else if (scores[i] < 15) { countBelow15++; System.out.println(names[i] + " has a score below 15: " + scores[i]); } // Sum the scores for average calculation totalScore += scores[i]; } // Calculate and display the average score double averageScore = totalScore / 10.0; System.out.println("\nAverage score: " + averageScore); System.out.println("Number of scores above 17: " + countAbove17); System.out.println("Number of scores below 15: " + countBelow15); scanner.close(); } }
